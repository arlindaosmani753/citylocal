**middleware.ts** — Use the exact pattern from 01-RESEARCH.md Pattern 1 (Three-Client Architecture). Critical requirements:
- Call `supabase.auth.getUser()` (NOT `getSession()`) — this validates the JWT against Supabase servers
- The `setAll` cookie handler MUST update BOTH `request.cookies` AND `supabaseResponse.cookies` — if either is missing, sessions break silently
- Protected routes: redirect to /login if user is null AND path does not start with `/login`, `/register`, `/auth`, `/cities`, or `/_next`
- The matcher config must exclude static assets: `['/((?!_next/static|_next/image|favicon.ico|.*\\.(?:svg|png|jpg|jpeg|gif|webp)$).*)']`

```typescript
// middleware.ts
import { createServerClient } from '@supabase/ssr'
import { NextResponse, type NextRequest } from 'next/server'

export async function middleware(request: NextRequest) {
  let supabaseResponse = NextResponse.next({ request })

  const supabase = createServerClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_PUBLISHABLE_KEY!,
    {
      cookies: {
        getAll() { return request.cookies.getAll() },
        setAll(cookiesToSet) {
          cookiesToSet.forEach(({ name, value }) => request.cookies.set(name, value))
          supabaseResponse = NextResponse.next({ request })
          cookiesToSet.forEach(({ name, value, options }) =>
            supabaseResponse.cookies.set(name, value, options))
        },
      },
    }
  )

  const { data: { user } } = await supabase.auth.getUser()

  const path = request.nextUrl.pathname
  const isPublicPath = path.startsWith('/login') ||
    path.startsWith('/register') ||
    path.startsWith('/auth') ||
    path.startsWith('/cities') ||
    path.startsWith('/profile')

  if (!user && !isPublicPath) {
    const url = request.nextUrl.clone()
    url.pathname = '/login'
    return NextResponse.redirect(url)
  }

  return supabaseResponse
}

export const config = {
  matcher: ['/((?!_next/static|_next/image|favicon.ico|.*\\.(?:svg|png|jpg|jpeg|gif|webp)$).*)'],
}
```

**src/actions/auth.ts** — Server Actions only (no client code). Five exports:

1. `registerUser(formData: FormData)` — Zod schema: email (z.string().email()), password (z.string().min(8).max(72)), username (z.string().min(3).max(50).regex(/^[a-z0-9_-]+$/)), homeCityId (z.string().uuid().optional()). On Zod failure → return `{ error: fieldErrors }`. On Supabase signUp failure → return `{ error: { _form: [message] } }`. On success → `db.insert(profiles)` with the same UUID from auth.users. Use `NEXT_PUBLIC_SITE_URL/auth/callback` as the emailRedirectTo. Redirect to `/verify-email` on success.

2. `signIn(formData: FormData)` — Zod: email + password. Call `supabase.auth.signInWithPassword()`. On error → return `{ error: { _form: [message] } }`. On success → `redirect('/')`.

3. `signOut()` — Call `supabase.auth.signOut()`. Redirect to `/login`.

4. `resetPassword(formData: FormData)` — Zod: email. Call `supabase.auth.resetPasswordForEmail(email, { redirectTo: ${siteUrl}/auth/callback?type=recovery })`. Return `{ success: true }` or `{ error: { _form: [message] } }`. Do NOT redirect — show confirmation in the form.

5. `updatePassword(formData: FormData)` — Zod: password (z.string().min(8).max(72)), confirm (same). Validate they match. Call `supabase.auth.updateUser({ password })`. Redirect to `/login` on success.
